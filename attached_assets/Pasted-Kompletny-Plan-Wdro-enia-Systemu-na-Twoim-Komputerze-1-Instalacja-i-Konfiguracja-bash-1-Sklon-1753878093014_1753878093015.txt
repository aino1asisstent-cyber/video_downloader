4. Kopia Bezpieczeństwa
Automatyczny backup kodu:

python
# backup.py
import os
import shutil
import datetime
from pathlib import Path

def create_backup():
    source_dir = Path(__file__).parent
    backup_dir = Path.home() / "backups" / "ai_file_manager"
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    
    backup_dir.mkdir(parents=True, exist_ok=True)
    backup_name = f"backup_{timestamp}.zip"
    
    shutil.make_archive(backup_dir / backup_name, 'zip', source_dir)
    return backup_dir / f"{backup_name}.zip"

if __name__ == "__main__":
    backup_path = create_backup()
    print(f"Utworzono backup: {backup_path}")
Automatyczny backup danych:

python
# data_backup.py
import shutil
import schedule
import time
from pathlib import Path

def backup_data():
    source_dirs = [
        Path.home() / "zip_downloads",
        Path.home() / "chat_history",
        Path.home() / "Videos"
    ]
    
    backup_dir = Path.home() / "backups" / "ai_data"
    backup_dir.mkdir(parents=True, exist_ok=True)
    
    for source in source_dirs:
        if source.exists():
            shutil.copytree(source, backup_dir / source.name, 
                            dirs_exist_ok=True)
    
    print(f"Zaktualizowano backup danych: {backup_dir}")

# Uruchamiaj codziennie o 2:00
schedule.every().day.at("02:00").do(backup_data)

while True:
    schedule.run_pending()
    time.sleep(60)

6. Testy Przed Deployem
test_app.py:

python
import unittest
import os
from app import VideoDownloader

class TestAppDeployment(unittest.TestCase):
    def test_ffmpeg_availability(self):
        """Test czy FFmpeg jest dostępny"""
        downloader = VideoDownloader()
        self.assertIsNotNone(downloader.ffmpeg_path, "FFmpeg nie jest zainstalowany")
        
    def test_directory_access(self):
        """Test uprawnień do zapisu w folderach"""
        test_dirs = [
            os.path.expanduser("~/Videos"),
            os.path.expanduser("~/Downloads"),
            os.path.expanduser("~/backups")
        ]
        
        for directory in test_dirs:
            test_file = os.path.join(directory, "test_file.txt")
            try:
                with open(test_file, "w") as f:
                    f.write("test")
                os.remove(test_file)
            except Exception as e:
                self.fail(f"Brak uprawnień do {directory}: {str(e)}")
    
    def test_replit_config(self):
        """Test konfiguracji Replit"""
        os.environ["REPLIT"] = "true"
        from app import get_download_dir
        download_dir = get_download_dir()
        self.assertTrue(download_dir.endswith("/downloads"))
        self.assertTrue(os.path.exists(download_dir))
        del os.environ["REPLIT"]

if __name__ == "__main__":
    unittest.main()
